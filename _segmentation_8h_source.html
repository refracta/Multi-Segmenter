196  float dv;

198  if (v < vmin)
199  v = vmin;

202  dv = vmax - vmin;
203 
204  if (v < (vmin + 0.25 * dv)) {

207  } else if (v < (vmin + 0.5 * dv)) {

210  } else if (v < (vmin + 0.75 * dv)) {
211  c.r = 4 * (v - vmin - 0.5 * dv) / dv;

213  } else {
