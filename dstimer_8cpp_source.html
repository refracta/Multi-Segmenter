188 void DS_timer::offTimer(UINT id) {
189  if (turnOn == false)
190  return;
191 
192  if (timerStates[id] == TIMER_OFF)
193  return;
194 
195 #ifdef _WIN32
196  QueryPerformanceCounter(&end_ticks[id]);
197  totalTicks[id].QuadPart = totalTicks[id].QuadPart + (end_ticks[id].QuadPart - start_ticks[id].QuadPart);
198 #else
199  gettimeofday(&end_ticks[id], NULL);
200  TIME_VAL period, previous;
201  timersub(&end_ticks[id], &start_ticks[id], &period);
202  previous = totalTicks[id];
203  timeradd(&previous, &period, &totalTicks[id]);
204 #endif
205 
206  timerStates[id] = TIMER_OFF;
207 }
